== String
[source,python]
----
# You can store a mixture of numbers and letters as strings by putting them in quotation marks
s1 = "abc"
s2 = "def123"

# You can use either single/double quotation
s1 = 'abc'
s2 = "abc"

# Numbers inside quotes are considered to be strings, even if they look like numbers
s3 = "123"

# You can add 2 strings together
"abc" + "def"

# You can convert numbers to string, or number-like strings into numbers
x = str(123)
y = int("123")
type(x)
type(y)

# You can convert all the characters in a string from lower case to upper case or vice versa
sequence = "aTCG"
sequence.upper()
sequence.lower()

== Functions 

[source,python]
----
# You can make a function to bundle a set of instructions
# This is often what you're using when you use tools developed by others
# Learning about how you make your own function can help 
# understanding how to use functions developed by other people.

# Create a simple function that prints a message
def say_something(x):
    out = "Say " + x
    return(out)

say_something("hey")

# You can define parameters for a function
# The first parameter in a function are often defined as x, 
# but you can define it with other variable names tools
def say_something(something):
    out = "Say " + something
    return(out)

# You can use the function you have defined
# The values you give the function are called "arguements" (e.g. "hey")
say_something("hey")

# You can have more than 1 value
def say_something(x, y):
    out = "Say " + x + ", and then " + y
    return(out)
say_something("hey", "there")

# By default the function will assume the argumnts are entered in order
# (i.e. the first value is the first variable, the second value is the second variable)
# However, you can also specify your arguments
say_something(x = "1st", y = "2nd")

# Notice that when you specify your arguments, it will assign the value to the
# correct parameters aeven when you can enter your arguments "out of order"
# This is why when you read tutorials, often arguements are not specified
say_something(y = "2nd", x = "1st")


# You can have pre-defined value for your parameters
def say_something(x = "nothing"):
    out = "Say " + x
    return(out)

# Then you don't have to put any arguements to call the function
say_something()

=== Simple loop
[source,python]
----
# Loops are very useful for automation
# It allows for a block of code to be repeated. 
# For example, imagine you have 3 files in a folder
# You can use for loop to perfom the same code on each of them
files = ["file1.txt", "file2.txt", "file3.txt"]

for current_file in files:
    # We're just printing the file name here,
    # In more complexed code, you can do a variety of things here
    # For example, you might want to find out the average of a specific column in all files in a folder.
    print(current_file)

# You can loop through a series of numbers created by  "range" 
for idx in range(len(files)):
    print(idx)

# You can combine this with what we learn about indexes to refer to specific values
for idx in range(len(files)):
    print(files[idx])
